96//----private------------------------------------------------------------------
97bool GraphicDisplay::OnInit()
98{
99 esp_err_t ret;
100
101 CORE_INFO("Initializing SPI bus");
102
103 spi_bus_config_t buscfg =
104 {
105 .mosi_io_num = _mosiPin,
106 .miso_io_num = _misoPin,
107 .sclk_io_num = _clkPin,
108 .quadwp_io_num = -1,
109 .quadhd_io_num = -1,
110 .max_transfer_sz = (DISP_H_RES * 80 * sizeof(uint16_t)),
111 };
112
113 // Choosing DMA channel 0 (no DMA) for simplicity; can be changed to SPI_DMA_CH_AUTO if needed
114 ret = spi_bus_initialize((spi_host_device_t)DISP_SPI_HOST, &buscfg, SPI_DMA_CH_AUTO);
115 ESP_ERROR_CHECK(ret);
116
117 // Initialize LVGL port
118 CORE_INFO("Initializing LVGL port");
119
120 lvgl_port_cfg_t lvgl_cfg = ESP_LVGL_PORT_INIT_CONFIG();
121 ret = lvgl_port_init(&lvgl_cfg);
122 ESP_ERROR_CHECK(ret);
123
124 // Configure and install the ILI9341 panel driver
125 CORE_INFO("Installing ILI9341 panel driver");
126
127 esp_lcd_panel_io_spi_config_t io_config =
128 {
129 .cs_gpio_num = _csPin,
130 .dc_gpio_num = _dcPin,
131 .spi_mode = 0,
132 .pclk_hz = 40 * 1000 * 1000, // 40 MHz clock
133 .trans_queue_depth = 10,
134 .on_color_trans_done = NULL,
135 .user_ctx = NULL,
136 .lcd_cmd_bits = 8,
137 .lcd_param_bits = 8,
138 .flags = { .dc_low_on_data = 0 }
139 };
140
141 // Create SPI IO handle for the panel
142 esp_lcd_panel_io_handle_t io_handle = nullptr;
143 ret = esp_lcd_new_panel_io_spi((esp_lcd_spi_bus_handle_t)DISP_SPI_HOST, &io_config, &io_handle);
144 ESP_ERROR_CHECK(ret);
145
146 esp_lcd_panel_dev_config_t panel_config =
147 {
148 .reset_gpio_num = _rstPin,
149 .rgb_ele_order = LCD_RGB_ELEMENT_ORDER_RGB,
150 .bits_per_pixel = 16,
151 };
152
153 // Create the panel handle
154 ret = esp_lcd_new_panel_ili9341(io_handle, &panel_config, &_panel_handle);
155 ESP_ERROR_CHECK(ret);
156
157 // Intialize the panel
158 ret = esp_lcd_panel_reset(_panel_handle);
159 ESP_ERROR_CHECK(ret);
160
161 ret = esp_lcd_panel_init(_panel_handle);
162 ESP_ERROR_CHECK(ret);
163
164 // Additional panel configurations
165 CORE_INFO("Configuring panel orientation and color settings");
166
167 ret = esp_lcd_panel_mirror(_panel_handle, false, true);
168 ESP_ERROR_CHECK(ret);
169
170 ret = esp_lcd_panel_swap_xy(_panel_handle, false);
171 ESP_ERROR_CHECK(ret);
172
173 // Turn on the display
174 ret = esp_lcd_panel_disp_on_off(_panel_handle, true);
175 ESP_ERROR_CHECK(ret);
176
177 // Add display to LVGL
178 CORE_INFO("Adding display to LVGL");
179
180 lvgl_port_display_cfg_t disp_cfg =
181 {
182 .io_handle = io_handle,
183 .panel_handle = _panel_handle,
184 .buffer_size = DISP_H_RES * 60,
185 .double_buffer = true,
186 .hres = DISP_H_RES,
187 .vres = DISP_V_RES,
188 .monochrome = false,
189 .rotation = { .swap_xy = false, .mirror_x = false, .mirror_y = true },
190 .flags = { .buff_dma = 1, .swap_bytes = 1 }
191 };
192
193 _lvgl_disp = lvgl_port_add_disp(&disp_cfg);
194
195 // Default background white so boot sequence is: white -> splash (white + logo) -> main UI
196 if (lvgl_port_lock(portMAX_DELAY))
197 {
198 lv_obj_t* scr = lv_scr_act();
199 lv_obj_set_style_bg_color(scr, lv_color_white(), LV_PART_MAIN);
200 lv_obj_set_style_bg_opa(scr, LV_OPA_COVER, LV_PART_MAIN);
201 lvgl_port_unlock();
202 }
203
204 CORE_INFO("Initializing XPT2046 touch driver");
205
206 esp_lcd_panel_io_handle_t tp_io_handle = nullptr;
207
208 esp_lcd_panel_io_spi_config_t tp_io_config =
209 {
210 .cs_gpio_num = _touchCsPin,
211 .dc_gpio_num = GPIO_NUM_NC,
212 .spi_mode = 0,
213 .pclk_hz = 1 * 1000 * 1000,
214 .trans_queue_depth = 10,
215 .on_color_trans_done = NULL,
216 .user_ctx = NULL,
217 .lcd_cmd_bits = 8,
218 .lcd_param_bits = 8,
219 .flags = {}
220 };
221
222 // Create SPI IO handle for touch panel
223 ret = esp_lcd_new_panel_io_spi((spi_host_device_t)DISP_SPI_HOST, &tp_io_config, &tp_io_handle);
224 ESP_ERROR_CHECK(ret);
225
226 esp_lcd_touch_config_t tp_cfg =
227 {
228 .x_max = DISP_H_RES,
229 .y_max = DISP_V_RES,
230 .rst_gpio_num = GPIO_NUM_NC,
231 .int_gpio_num = (gpio_num_t)_touchIrqPin,
232 .levels = { .reset = 0, .interrupt = 0 },
233 .flags = { .swap_xy = false, .mirror_x = false, .mirror_y = true }
234 };
235
236 // Create touch handle
237 ret = esp_lcd_touch_new_spi_xpt2046(tp_io_handle, &tp_cfg, &_tp_handle);
238 ESP_ERROR_CHECK(ret);
239
240 // Add touch to LVGL
241 CORE_INFO("Add touch to LVGL");
242
243 lvgl_port_touch_cfg_t touch_cfg =
244 {
245 .disp = _lvgl_disp,
246 .handle = _tp_handle,
247 };
248 lvgl_port_add_touch(&touch_cfg);
249
250 CORE_INFO("Finalizing setup...");
251
252 if (lvgl_port_lock(portMAX_DELAY))
253 {
254 SetupTouchDetection();
255
256 lvgl_port_unlock();
257 }
258
259 // Initialize custom UI (generated code); then ensure splash is shown first
260 if (lvgl_port_lock(portMAX_DELAY))
261 {
262 ui_init();
263 lv_disp_load_scr(ui_SplashScreen);
264 lvgl_port_unlock();
265 }
266
267 _valid = true;
268 return true;
269}

301void GraphicDisplay::OnTouchEventCallback(lv_event_t* e)
302{
303 lv_event_code_t code = lv_event_get_code(e);
304
305 if (code == LV_EVENT_CLICKED)
306 {
307 GraphicDisplay* self = GraphicDisplay::GetInstance();
308 const uint32_t now = lv_tick_get();
309
310 if (self->_lastTapTime != 0 && (now - self->_lastTapTime) < DOUBLE_TAP_MS)
311 {
312 CORE_INFO(">>> Double tap detected <<<");
313 self->_lastTapTime = 0;
314
315 if (self->_onDoubleClickAction != nullptr)
316 {
317 self->_onDoubleClickAction();
318 }
319 }
320 else
321 {
322 self->_lastTapTime = now;
323 }
324 }
325 else if (code == LV_EVENT_LONG_PRESSED)
326 {
327 CORE_INFO(">>> LV_EVENT_LONG_PRESSED detected <<<");
328
329 // Call the registered long press action
330 if (GraphicDisplay::GetInstance()->_onLongPressAction != nullptr)
331 {
332 GraphicDisplay::GetInstance()->_onLongPressAction();
333 }
334 }
335}

338void GraphicDisplay::SetupTouchDetection()
339{
340 lv_obj_t * top_layer = lv_layer_top();
341
342 lv_obj_add_flag(top_layer, LV_OBJ_FLAG_CLICKABLE);
343
344 lv_obj_add_event_cb(top_layer, OnTouchEventCallback, LV_EVENT_ALL, nullptr);
345
346 lv_indev_t* indev = nullptr;
347 while ((indev = lv_indev_get_next(indev)) != nullptr)
348 {
349 if (lv_indev_get_type(indev) == LV_INDEV_TYPE_POINTER)
350 {
351 lv_indev_set_long_press_time(indev, 1400);
352 break;
353 }
354 }
355
356 CORE_INFO("Touch detection setup completed on TOP LAYER.");
357}
