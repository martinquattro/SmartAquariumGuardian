30void FoodFeeder::OnUpdate()
31{
32 Utils::DateTime currentTime;
33
34 if (!Core::GuardianProxy::GetInstance()->GetDateTime(currentTime))
35 {
36 CORE_ERROR("Failed to get current time. System cannot get updated");
37 return;
38 }
39
40 const int currentMinute = currentTime.ToMinutesOfDay();
41
42 // Check feeding schedule
43 if (currentMinute != _lastFeedTime)
44 {
45 const auto& scheduleList = Core::GuardianProxy::GetInstance()->GetFeedingScheduleFromStorage();
46
47 for (const auto& entry : scheduleList)
48 {
49 if (entry._enabled && entry._min == currentMinute)
50 {
51 CORE_INFO("Scheduled feeding triggered for slot %d: Dose=%d at %d minutes after midnight.",
52 entry._id, entry._dose, entry._min);
53
54 const Result feedResult = this->Feed(entry._dose);
55 if (!feedResult.success)
56 {
57 CORE_ERROR("Scheduled feeding failed: %s", feedResult.responseMessage.value_or("Unknown error").c_str());
58 }
59 else
60 {
61 _lastFeedTime = currentMinute;
62 CORE_INFO("Scheduled feeding started successfully for slot %d.", entry._id);
63 }
64 }
65 }
66 }
67}

70auto FoodFeeder::Feed(int dose) -> Result
71{
72 CORE_INFO("Feeding %d doses ", dose);
73
74 if (dose < MIN_FEED_DOSE || dose > MAX_FEED_DOSE)
75 {
76 CORE_WARNING("Invalid dose %d. Must be in range [%d, %d].", dose, MIN_FEED_DOSE, MAX_FEED_DOSE);
77 return Result::Error("Invalid dose amount.");
78 }
79
80 if (_feedingTask.IsBusy())
81 {
82 CORE_WARNING("Feeding already in progress.");
83 return Result::Error("Feeding in progress.");
84 }
85
86 bool started = _feedingTask.Start(
87 [this, dose]()
88 {
89 this->PerformAsyncFeedingSequence(dose);
90 }
91 );
92
93 if (!started)
94 {
95 return Result::Error("Internal Error: Failed to start feeding.");
96 }
97
98 return Result::Success("Feeding process started");
99}

168auto FoodFeeder::GetFeederStatus() const -> FeederStatus
169{
170 FeederStatus status;
171
172 const auto& scheduleList = Core::GuardianProxy::GetInstance()->GetFeedingScheduleFromStorage();
173 Utils::DateTime currentTime;
174 if (!Core::GuardianProxy::GetInstance()->GetDateTime(currentTime))
175 {
176 CORE_ERROR("Failed to get current time. Status cannot be retrieved");
177 return status;
178 }
179
180 const int currentMinutes = currentTime.ToMinutesOfDay();
181
182 status.totalPerDay = 0;
183 status.remainingDosesToday = 0;
184 status.nextFeedDoses = 0;
185 status.nextFeedTime = Utils::DateTime(0, 0, 0); // Default to midnight
186
187 int earliestNextFeedMinutes = MINUTES_IN_A_DAY + 1;
188
189 for (const auto& entry : scheduleList)
190 {
191 if (entry._enabled)
192 {
193 status.totalPerDay += entry._dose;
194
195 if (entry._min > currentMinutes)
196 {
197 status.remainingDosesToday += entry._dose;
198
199 if (entry._min < earliestNextFeedMinutes)
200 {
201 earliestNextFeedMinutes = entry._min;
202 status.nextFeedDoses = entry._dose;
203 }
204 }
205 }
206 }
207
208 if (earliestNextFeedMinutes <= MINUTES_IN_A_DAY)
209 {
210 status.nextFeedTime = Utils::DateTime(earliestNextFeedMinutes * 60);
211 }
212
213 return status;
214}

216//----private------------------------------------------------------------------
217void FoodFeeder::PerformAsyncFeedingSequence(int dose)
218{
219 CORE_INFO("Feeding sequence started for %d doses.", dose);
220
221 Core::GuardianProxy::GetInstance()->UpdateFeedingStatusIndicator(true);
222
223 Drivers::Servo* servo = Drivers::Servo::GetInstance();
224
225 for (int i = 0; i < dose; ++i)
226 {
227 CORE_INFO("Dispensing dose %d of %d.", i + 1, dose);
228
229 // Open feeder
230 servo->FadeToAngle(FEEDER_OPEN_ANGLE, FEEDER_MOVE_TIME_MS);
231 TaskDelayMs(FEEDER_MOVE_TIME_MS + FEEDER_WAIT_TIME_MS);
232
233 // Close feeder
234 servo->FadeToAngle(FEEDER_CLOSED_ANGLE, FEEDER_MOVE_TIME_MS);
235 TaskDelayMs(FEEDER_MOVE_TIME_MS + FEEDER_WAIT_TIME_MS);
236
237 servo->Release();
238 }
239
240 CORE_INFO("Feeding sequence completed for %d doses.", dose);
241 Core::GuardianProxy::GetInstance()->UpdateFeedingStatusIndicator(false);
242}
