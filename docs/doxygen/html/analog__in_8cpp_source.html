56float AnalogIn::ReadVoltage(const int samples) const
57{
58 if (!_valid) return 0;
59
60 int raw = 0;
61 int sumRaw = 0;
62 int count = 0;
63 for (size_t i = 0; i < samples; i++)
64 {
65 if (adc_oneshot_read(_handle, _channel, &raw) == ESP_OK)
66 {
67 sumRaw += raw;
68 count++;
69 }
70 }
71
72 const float avgRaw = static_cast<float>(sumRaw) / count;
73
74 int mv = 0;
75 if (adc_cali_raw_to_voltage(_cali, avgRaw, &mv) == ESP_OK)
76 {
77 return mv / 1000.0f;
78 }
79
80 const int max_raw = (1 << _width) - 1;
81 float volts = (static_cast<float>(avgRaw) / static_cast<float>(max_raw)) * _vref;
82
83 return static_cast<int>(volts * 1000.0f + 0.5f);
84}
