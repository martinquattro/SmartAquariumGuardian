21class JsonPayloadParser
22{
23 public:
24
25 explicit JsonPayloadParser(const std::string& payload)
26 : _isValid(false)
27 {
28 // 1. Quick check for valid JSON syntax using accept()
29 if (!Json::accept(payload))
30 {
31 CORE_ERROR("JsonPayloadParser: Invalid JSON format (accept failed)");
32 _errorMsg = "Invalid JSON format.";
33 return;
34 }
35
36 // 2. Parse without throwing exceptions.
37 // The third argument 'false' disables exception throwing.
38 // The second argument 'nullptr' is for a parse event callback (not used here).
39 _json = Json::parse(payload, nullptr, false);
40
41 // 3. Check if parsing was successful using .is_discarded()
42 if (_json.is_discarded())
43 {
44 CORE_ERROR("JsonPayloadParser: Parse error (is_discarded)");
45 _errorMsg = "Internal error parsing JSON structure.";
46 _isValid = false;
47 }
48 else
49 {
50 // Parsing successful
51 _isValid = true;
52 }
53 }
54
55 bool IsValid() const { return _isValid; }
56

76template<typename T>
77std::optional<T> JsonPayloadParser::GetValue(const std::string& key, const std::string& parentKey) const
78{
79 if (!_isValid)
80 {
81 return std::nullopt;
82 }
83
84 // 1. Determine the root object to search in
85 const Json* rootPtr = &_json;
86 if (!parentKey.empty())
87 {
88 auto itParent = _json.find(parentKey);
89 if (itParent == _json.end() || !itParent->is_object())
90 {
91 // Parent key doesn't exist or isn't an object
92 return std::nullopt;
93 }
94 rootPtr = &(*itParent);
95 }
96 const Json& root = *rootPtr;
97
98 // 2. Find the key using .find() (safe for const objects)
99 auto it = root.find(key);
100 if (it == root.end())
101 {
102 // Key not found
103 return std::nullopt;
104 }
105 const Json& valueJson = *it;
106
107 // 3. Check type and extract safely without exceptions
108 // We use 'if constexpr' so the compiler only generates code for the matching type branch.
109 if constexpr (std::is_same_v<T, float>)
110 {
111 if (valueJson.is_number())
112 {
113 return valueJson.get<float>(); // Safe conversion
114 }
115
116 if (valueJson.is_null())
117 {
118 return 0.0f;
119 }
120 }
121 else if constexpr (std::is_same_v<T, int>)
122 {
123 if (valueJson.is_number_integer())
124 {
125 return valueJson.get<int>(); // Safe conversion
126 }
127
128 if (valueJson.is_null())
129 {
130 return 0;
131 }
132 }
133 else if constexpr (std::is_same_v<T, bool>)
134 {
135 if (valueJson.is_boolean())
136 {
137 return valueJson.get<bool>();
138 }
139 }
140 else if constexpr (std::is_same_v<T, std::string>)
141 {
142 if (valueJson.is_string())
143 {
144 return valueJson.get<std::string>();
145 }
146 }
147
148 // If type mismatch found
149 CORE_WARNING("JsonPayloadParser: Type mismatch for key '%s'.", key.c_str());
150 return std::nullopt;
151}
