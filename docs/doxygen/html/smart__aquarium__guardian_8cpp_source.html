22bool SmartAquariumGuardian::OnInit()
23{
24 // Create and initialize the GuardianProxy
25 Core::GuardianProxy::GetInstance()->Init();
26
27 // Display + backlight draw high current; delay after before starting network
28 Managers::UserInterface::GetInstance()->Init(300);
29
30 // Initialize services
31 Services::RealTimeClock::GetInstance()->Init(100);
32 Services::StorageService::GetInstance()->Init(100);
33 Services::PowerController::GetInstance()->Init(100);
34
35 // Initialize managers (sensors first, low current)
36 Managers::WaterMonitor::GetInstance()->Init(200);
37
38 // Servo init can draw significant current; delay after so supply can settle
39 Managers::FoodFeeder::GetInstance()->Init(400);
40
41 Managers::NetworkController::GetInstance()->Init();
42
43 _delay.Start(Config::SYSTEM_TIME_INCREMENT_MS);
44
45 return true;
46}
