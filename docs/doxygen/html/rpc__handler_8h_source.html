42 Result Handle(const std::string& payload) override

44 Utils::JsonPayloadParser parser(payload);
45 if (!parser.IsValid())
46 {
47 return Result::Error("Internal error: Invalid JSON payload.");

49
50 const auto minEnabledOpt = parser.GetParam<bool>(NetworkConfig::ClientAttributes::TEMP_LIMIT_MIN_ENABLED);
51 const auto maxEnabledOpt = parser.GetParam<bool>(NetworkConfig::ClientAttributes::TEMP_LIMIT_MAX_ENABLED);
52
53 if (!minEnabledOpt.has_value() || !maxEnabledOpt.has_value())
54 {
55 return Result::Error("Missing required 'enabled' flags for limits.");
56 }
57
58 bool minEnabled = minEnabledOpt.value();
59 bool maxEnabled = maxEnabledOpt.value();
60
61 float tempMin = 0.0f;
62 float tempMax = 0.0f;
63
64 const auto tempMinOpt = parser.GetParam<float>(NetworkConfig::ClientAttributes::TEMP_LIMIT_MIN);
65 const auto tempMaxOpt = parser.GetParam<float>(NetworkConfig::ClientAttributes::TEMP_LIMIT_MAX);
66
67 if (minEnabled)
68 {
69 if (!tempMinOpt.has_value())
70 {
71 return Result::Error("Min limit is enabled but value is invalid or missing (cannot be null).");
72 }
73
74 tempMin = tempMinOpt.value();
75 }
76 else
77 {
78 if (tempMinOpt.has_value())
79 tempMin = tempMinOpt.value();
80 }
