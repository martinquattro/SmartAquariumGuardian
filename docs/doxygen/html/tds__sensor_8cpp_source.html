25void TdsSensor::OnUpdate()
26{
27 const float analogReading = _analogInPin.ReadVoltage(64);
28 const float avgAnalogReading = StoreAnalogReading(analogReading);
29
30 // Logic to transform reading to ppm units
31 // ppm = (133.42 * V³ - 255.86 * V² + 857.39 * V) / factorTemp * 0.5
32 // factorTemp ≈ 1.0 + 0.02 * (temp - 25)
33 float factorTemp = (1.0 + 0.02 * (_temperature - 25.0));
34
35 _lastReading = (133.42 * pow(avgAnalogReading, 3)
36 - 255.86 * pow(avgAnalogReading, 2)
37 + 857.39 * avgAnalogReading) / factorTemp * 0.5;
38
39 _lastReading = std::clamp(_lastReading, MIN_TDS_VALUE, MAX_TDS_VALUE);
40
41 CORE_INFO("Tds avg reading - PPM = %d", _lastReading);
42}

52float TdsSensor::StoreAnalogReading(const float reading)
53{
54 if (reading < 0.0f || reading > 3.3f)
55 {
56 CORE_ERROR("Invalid analog reading: %.4f", reading);
57 return 0.0f;
58 }
59
60 // Add reading to vector
61 (*_analogReadingsVecIter) = reading;
62
63 // Check if we were at last element of the readings vector
64 if (_analogReadingsVecIter++ >= _analogReadingsVec.end())
65 {
66 _analogReadingsVecIter = _analogReadingsVec.begin();
67 }
68
69 // Obtain average
70 float tdsReadingSum = 0.0;
71 int amountOfReadings = 0;
72 for (auto it = _analogReadingsVec.begin(); it != _analogReadingsVec.end(); ++it)
73 {
74 if ((*it) > -1.0f)
75 {
76 tdsReadingSum += (*it);
77 ++amountOfReadings;
78 }
79 }
80
81 float avgAnalogReading = (tdsReadingSum / amountOfReadings);
82 return avgAnalogReading;
83}
