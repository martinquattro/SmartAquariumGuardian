49//----private------------------------------------------------------------------
50int16_t TemperatureSensor::GetRawReading()
51{
52 if (!_oneWirePin.Reset())
53 {
54 CORE_ERROR("No DS18B20 detected!");
55 return 0;
56 }
57
58 _oneWirePin.WriteByte(CMD_SKIP_ROM);
59 _oneWirePin.WriteByte(CMD_CONVERT_T);
60
61 // Wait for conversion (750 ms for 12-bit resolution)
62 TaskDelayMs(750);
63
64 if (!_oneWirePin.Reset())
65 {
66 CORE_ERROR("Lost device after conversion!");
67 return 0;
68 }
69
70 _oneWirePin.WriteByte(CMD_SKIP_ROM);
71 _oneWirePin.WriteByte(CMD_READ_SCRATCH);
72
73 uint8_t scratchpad[9];
74 for (int i = 0; i < 9; i++)
75 {
76 scratchpad[i] = _oneWirePin.ReadByte();
77 }
78
79 // Verify CRC
80 uint8_t crcCalculated = CalculateCRC8(scratchpad, 8);
81 if (crcCalculated != scratchpad[8])
82 {
83 CORE_ERROR("DS18B20 CRC check failed! Calculated: 0x%02X, Received: 0x%02X", crcCalculated, scratchpad[8]);
84 return -1;
85 }
86
87 uint8_t tempLSB = scratchpad[0];
88 uint8_t tempMSB = scratchpad[1];
89
90 int16_t rawTemp = (tempMSB << 8) | tempLSB;
91 return rawTemp;
92}

95float TemperatureSensor::StoreReading(float reading)
96{
97 (*_rawReadingsVecIter) = reading;
98
99 if (++_rawReadingsVecIter >= _rawReadingsVec.end())
100 {
101 _rawReadingsVecIter = _rawReadingsVec.begin();
102 }
103
104 float sum = 0.0f;
105 int count = 0;
106 for (auto val : _rawReadingsVec)
107 {
108 if (val > -1.0f)
109 {
110 sum += val;
111 count++;
112 }
113 }
114
115 return (count > 0) ? (sum / count) : reading;
116}

118//----private------------------------------------------------------------------
119uint8_t TemperatureSensor::CalculateCRC8(const uint8_t *addr, uint8_t len)
120{
121 uint8_t crc = 0;
122
123 for (uint8_t i = 0; i < len; i++)
124 {
125 uint8_t inbyte = addr[i];
126 for (uint8_t j = 0; j < 8; j++)
127 {
128 uint8_t mix = (crc ^ inbyte) & 0x01;
129 crc >>= 1;
130 if (mix)
131 {
132 crc ^= 0x8C;
133 }
134 inbyte >>= 1;
135 }
136 }
137
138 return crc;
139}
