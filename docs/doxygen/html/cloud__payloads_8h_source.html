49/*!
50 * @brief Builds JSON payload for client attributes published to v1/devices/me/attributes.

53class ClientAttributesPayload
54{
55 public:
56
57 ClientAttributesPayload()

60
61 _timezone = proxy->GetTimezoneFromStorage();

64 _scheduleList = proxy->GetFeedingScheduleFromStorage();
65 _wifiSsid = proxy->GetWifiSsid();
66 _wifiRssi = proxy->GetWifiRssi();
67
68 Utils::DateTime dt;

71 _deviceTime = dt.ToString();

73 else
74 {
75 _deviceTime = "--:--";
76 }
77 }

79 std::string ToJsonString() const
80 {
81 Json doc;

83 doc[NetworkConfig::ClientAttributes::TIMEZONE] = _timezone;
84 doc[NetworkConfig::ClientAttributes::TEMP_LIMIT_MIN] = _minTemp;
85 doc[NetworkConfig::ClientAttributes::TEMP_LIMIT_MIN_ENABLED] = _minEnabled;
86 doc[NetworkConfig::ClientAttributes::TEMP_LIMIT_MAX] = _maxTemp;
87 doc[NetworkConfig::ClientAttributes::TEMP_LIMIT_MAX_ENABLED] = _maxEnabled;
88
89 doc[NetworkConfig::ClientAttributes::TDS_LIMIT_MIN] = _minTds;
90 doc[NetworkConfig::ClientAttributes::TDS_LIMIT_MIN_ENABLED] = _tdsMinEnabled;
91 doc[NetworkConfig::ClientAttributes::TDS_LIMIT_MAX] = _maxTds;
92 doc[NetworkConfig::ClientAttributes::TDS_LIMIT_MAX_ENABLED] = _tdsMaxEnabled;
93
94 Json scheduleArray = Json::array();
95 for (const auto& e : _scheduleList)
96 {
97 Json entry;
98 entry[NetworkConfig::ClientAttributes::FEED_SLOT_ID] = e._id;
99 entry[NetworkConfig::ClientAttributes::FEED_TIME] = e._min;
100 entry[NetworkConfig::ClientAttributes::FEED_DOSE] = e._dose;
101 entry[NetworkConfig::ClientAttributes::FEED_ENABLED] = e._enabled;
102 scheduleArray.push_back(entry);
103 }
104 doc[NetworkConfig::ClientAttributes::FEEDING_SCHEDULE] = scheduleArray;
105
106 doc[NetworkConfig::ClientAttributes::WIFI_SSID] = _wifiSsid;
107 doc[NetworkConfig::ClientAttributes::WIFI_RSSI] = _wifiRssi;
108 doc[NetworkConfig::ClientAttributes::DEVICE_TIME] = _deviceTime;
109
110 return doc.dump();
111 }
