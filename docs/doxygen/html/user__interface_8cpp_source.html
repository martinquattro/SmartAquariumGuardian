74void UserInterface::OnUpdate()
75{
76 // Power status
77 {
78 UpdatePowerIndicator();
79 }
80
81 // Connection status
82 {
83 bool wifiOk = Core::GuardianProxy::GetInstance()->IsWifiConnected();
84 bool cloudOk = Core::GuardianProxy::GetInstance()->IsMqttConnected();
85 bool apPortalOk = Core::GuardianProxy::GetInstance()->IsApPortalActive();
86
87 if (!wifiOk)
88 {
89 if (apPortalOk)
90 {
91 _apIconOn->Show();
92 _wifiIconOff->Hide();
93 _wifiIconOn->Hide();
94 _cloudIconOff->Hide();
95 _cloudIconOn->Hide();
96 }
97 else
98 {
99 _apIconOn->Hide();
100 _wifiIconOff->Show();
101 _wifiIconOn->Hide();
102 _cloudIconOff->Show();
103 _cloudIconOn->Hide();
104 }
105 }
106 else if (!cloudOk)
107 {
108 _apIconOn->Hide();
109 _wifiIconOff->Hide();
110 _wifiIconOn->Show();
111 _cloudIconOff->Show();
112 _cloudIconOn->Hide();
113 }
114 else
115 {
116 _apIconOn->Hide();
117 _wifiIconOff->Hide();
118 _wifiIconOn->Show();
119 _cloudIconOff->Hide();
120 _cloudIconOn->Show();
121 }
122 }
123
124 // Time
125 {
126 Utils::DateTime dateTime;
127 if (Core::GuardianProxy::GetInstance()->GetDateTime(dateTime))
128 {
129 _time->SetText(dateTime.ToString().c_str());
130 }
131 }
132
133 // Temperature Panel
134 {
135 char buffer [50];
136
137 const float tempReading = Core::GuardianProxy::GetInstance()->GetTemperatureReading();
138 std::sprintf(buffer, "%.1f", tempReading);
139
140 _tempValue->SetText(buffer);
141
142 // Temperature limits
143 float minTemp = 0.0f, maxTemp = 0.0f;
144 bool isMinLimitEnabled = false, isMaxLimitEnabled = false;
145
146 Core::GuardianProxy::GetInstance()->GetTemperatureLimits(minTemp, isMinLimitEnabled, maxTemp, isMaxLimitEnabled);
147
148 if (isMinLimitEnabled)
149 {
150 std::sprintf(buffer, "%.1f °C", minTemp);
151 }
152 else
153 {
154 std::sprintf(buffer, "---");
155 }
156
157 _tempMinValue->SetText(buffer);
158
159 if (isMaxLimitEnabled)
160 {
161 std::sprintf(buffer, "%.1f °C", maxTemp);
162 }
163 else
164 {
165 std::sprintf(buffer, "---");
166 }
167
168 _tempMaxValue->SetText(buffer);
169
170 // Panel state
171 if (Core::GuardianProxy::GetInstance()->IsTemperatureOutOfLimits())
172 {
173 // Alert state
174 _tempPanel->SetState1();
175 }
176 else
177 {
178 // Normal state
179 _tempPanel->ClearState1();
180 }
181 }
182
183 // TDS Panel
184 {
185 char buffer [50];
186
187 const int tdsReading = Core::GuardianProxy::GetInstance()->GetTdsReading();
188 std::sprintf(buffer, "%d", tdsReading);
189
190 _tdsValue->SetText(buffer);
191
192 // TDS limits
193 int minTds = 0, maxTds = 0;
194 bool isMinLimitEnabled = false, isMaxLimitEnabled = false;
195
196 Core::GuardianProxy::GetInstance()->GetTdsLimits(minTds, isMinLimitEnabled, maxTds, isMaxLimitEnabled);
197
198 if (isMinLimitEnabled)
199 {
200 std::sprintf(buffer, "%d ppm", minTds);
201 }
202 else
203 {
204 std::sprintf(buffer, "---");
205 }
206
207 _tdsMinValue->SetText(buffer);
208
209 if (isMaxLimitEnabled)
210 {
211 std::sprintf(buffer, "%d ppm", maxTds);
212 }
213 else
214 {
215 std::sprintf(buffer, "---");
216 }
217
218 _tdsMaxValue->SetText(buffer);
219
220 // Panel state
221 if (Core::GuardianProxy::GetInstance()->IsTdsOutOfLimits())
222 {
223 // Alert state
224 _tdsPanel->SetState1();
225 }
226 else
227 {
228 // Normal state
229 _tdsPanel->ClearState1();
230 }
231 }
232
233 // Feeder
234 {
235 const auto& feederStatus = Core::GuardianProxy::GetInstance()->GetFeederStatus();
236
237 CORE_INFO("Feeder Status - Next Feed Time: %s, Next Feed Doses: %d, Remaining Doses Today: %d, Total Per Day: %d",
238 feederStatus.nextFeedTime.ToString().c_str(),
239 feederStatus.nextFeedDoses,
240 feederStatus.remainingDosesToday,
241 feederStatus.totalPerDay
242 );
243
244 // Next feeding time
245 char buffer [50];
246
247 if (feederStatus.remainingDosesToday > 0)
248 {
249 std::sprintf(buffer, "%s [%d]", feederStatus.nextFeedTime.ToString().c_str(), feederStatus.nextFeedDoses);
250 }
251 else if (feederStatus.totalPerDay > 0)
252 {
253 std::sprintf(buffer, "Tomorrow");
254 }
255 else
256 {
257 std::sprintf(buffer, "---");
258 }
259
260 _nextFeedingTime->SetText(buffer);
261
262 // Doses per day
263 std::sprintf(buffer, "%d", feederStatus.totalPerDay);
264 _dosesPerDay->SetText(buffer);
265
266 // Doses left today
267 std::sprintf(buffer, "%d", feederStatus.remainingDosesToday);
268 _dosesLeft->SetText(buffer);
269 }
270
271 if (!_firstUpdateDone)
272 {
273 lv_disp_load_scr(ui_Screen);
274 _display->SetBrightness(DISPLAY_BRIGHTNESS_NORMAL_MODE);
275 _firstUpdateDone = true;
276 }
277}

292void UserInterface::UpdatePowerIndicator()
293{
294 static Services::PowerController::BatteryLevel lastBatteryLevel = Services::PowerController::BatteryLevel::_size;
295 static Services::PowerController::Mode lastPowerMode = Services::PowerController::Mode::_size;
296
297 const auto powerMode = Core::GuardianProxy::GetInstance()->GetCurrentMode();
298 const auto batteryLevel = Core::GuardianProxy::GetInstance()->GetBatteryLevel();
299
300 if (lastBatteryLevel != batteryLevel || lastPowerMode != powerMode)
301 {
302 lastBatteryLevel = batteryLevel;
303 lastPowerMode = powerMode;
304
305 _batteryFullIcon->Hide();
306 _batteryHighIcon->Hide();
307 _batteryMediumIcon->Hide();
308 _batteryLowIcon->Hide();
309 _batteryCriticalIcon->Hide();
310
311 if (powerMode == Services::PowerController::Mode::MODE_BATTERY_POWERED)
312 {
313 switch (batteryLevel)
314 {
315 case Services::PowerController::BatteryLevel::LEVEL_FULL: _batteryFullIcon->Show(); break;
316 case Services::PowerController::BatteryLevel::LEVEL_HIGH: _batteryHighIcon->Show(); break;
317 case Services::PowerController::BatteryLevel::LEVEL_MEDIUM: _batteryMediumIcon->Show(); break;
318 case Services::PowerController::BatteryLevel::LEVEL_LOW: _batteryLowIcon->Show(); break;
319 case Services::PowerController::BatteryLevel::LEVEL_CRITICAL: _batteryCriticalIcon->Show(); break;
320 default: break;
321 }
322 }
323 }
324}
