26auto RealTimeClock::GetTime(Utils::DateTime& time) -> bool
27{
28 uint8_t startReg = 0x00;
29 uint8_t buffer[3] = {0};
30
31 if (!_i2c.WriteRead(&startReg, 1, buffer, 3))
32 {
33 CORE_ERROR("Failed to read time");
34 return false;
35 }
36
37 uint8_t seconds = BcdToDec(buffer[0]);
38 uint8_t minutes = BcdToDec(buffer[1]);
39 uint8_t hours = BcdToDec(buffer[2]);
40
41 time.Set(hours, minutes, seconds);
42
43 return true;
44}

46//-----------------------------------------------------------------------------
47bool RealTimeClock::SetTime(const Utils::DateTime& dateTime)
48{
49 uint8_t hours = dateTime.GetHour();
50 uint8_t minutes = dateTime.GetMinute();
51 uint8_t seconds = dateTime.GetSecond();
52
53
54 uint8_t buffer[4];
55 buffer[0] = 0x00; // start register
56 buffer[1] = DecToBcd(seconds);
57 buffer[2] = DecToBcd(minutes);
58 buffer[3] = DecToBcd(hours);
59
60 if (!_i2c.Write(buffer, sizeof(buffer)))
61 {
62 CORE_ERROR("Failed to set time");
63 return false;
64 }
65
66 CORE_INFO("Set RTC time to %02u:%02u:%02u", hours, minutes, seconds);
67 return true;
68}

102{
103 CORE_INFO("Time synchronized from NTP server");

105 time_t now;

115
