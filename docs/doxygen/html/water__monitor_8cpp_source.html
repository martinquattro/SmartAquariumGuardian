48Result WaterMonitor::SetTemperatureLimits(const float minTemp, const bool isMinLimitEnabled, const float maxTemp, const bool isMaxLimitEnabled)
49{
50 CORE_INFO("WaterMonitor: Request to set temp limits -> Min: %.2f (En:%d), Max: %.2f (En:%d)",
51 minTemp, isMinLimitEnabled, maxTemp, isMaxLimitEnabled);
52
53 if (isMinLimitEnabled && isMaxLimitEnabled)
54 {
55 if (minTemp >= maxTemp)
56 {
57 CORE_ERROR("WaterMonitor Validation Error: Min temp (%.2f) cannot be >= Max temp (%.2f) when both are enabled.", minTemp, maxTemp);
58 return Result::Error("Invalid limits: Minimum must be less than Maximum.");
59 }
60 }
61
62 if (isMinLimitEnabled)
63 {
64 if (minTemp < MIN_TEMP_VALID_VALUE || minTemp > MAX_TEMP_VALID_VALUE)
65 {
66 CORE_ERROR("WaterMonitor Validation Error: Min temp (%.2f) out of valid range [%.2f, %.2f].", minTemp, MIN_TEMP_VALID_VALUE, MAX_TEMP_VALID_VALUE);
67 return Result::Error("Invalid minimum temperature limit.");
68 }
69 }
70
71 if (isMaxLimitEnabled)
72 {
73 if (maxTemp < MIN_TEMP_VALID_VALUE || maxTemp > MAX_TEMP_VALID_VALUE)
74 {
75 CORE_ERROR("WaterMonitor Validation Error: Max temp (%.2f) out of valid range [%.2f, %.2f].", maxTemp, MIN_TEMP_VALID_VALUE, MAX_TEMP_VALID_VALUE);
76 return Result::Error("Invalid maximum temperature limit.");
77 }
78 }
79
80 const bool success = Core::GuardianProxy::GetInstance()->SaveTempLimitsInStorage(minTemp, isMinLimitEnabled, maxTemp, isMaxLimitEnabled);
81
82 if (success)
83 {
84 CORE_INFO("WaterMonitor: All temperature limit parameters saved successfully.");
85 return Result::Success("Temperature limits updated successfully.");
86 }
87 else
88 {
89 CORE_ERROR("WaterMonitor Storage Error: Failed to save temperature parameters.");
90 return Result::Error("Internal Error: Could not save settings to permanent memory.");
91 }
92}

100//-----------------------------------------------------------------------------
101bool WaterMonitor::IsTemperatureOutOfLimits() const
102{
103 float minTemp = 0.0f, maxTemp = 0.0f;
104 bool isMinLimitEnabled = false, isMaxLimitEnabled = false;
105
106 GetTemperatureLimits(minTemp, isMinLimitEnabled, maxTemp, isMaxLimitEnabled);
107
108 const float currentTemp = GetTemperatureReading();
109
110 if (isMinLimitEnabled && (currentTemp < minTemp))
111 {
112 CORE_INFO("Temperature reading %.2f is below the minimum limit of %.2f.", currentTemp, minTemp);
113 return true;
114 }
115
116 if (isMaxLimitEnabled && (currentTemp > maxTemp))
117 {
118 CORE_INFO("Temperature reading %.2f is above the maximum limit of %.2f.", currentTemp, maxTemp);
119 return true;
120 }
121
122 return false;
