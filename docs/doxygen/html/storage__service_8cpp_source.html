35bool StorageService::SaveFeedingScheduleInStorage(const int timeMinutesAfterMidnight, const int slotIndex, const int dose, const bool enabled)
36{
37 auto scheduleList = _configCache._feedingSchedule;
38
39 // Find existing entry by slotIndex
40 auto it = std::find_if(
41 scheduleList.begin(),
42 scheduleList.end(),
43 [slotIndex](const Services::FeedingScheduleEntry& entry)
44 {
45 return entry._id == slotIndex;
46 }
47 );
48
49 if (it != scheduleList.end())
50 {
51 // Update existing entry
52 it->_min = timeMinutesAfterMidnight;
53 it->_dose = dose;
54 it->_enabled = enabled;
55 }
56 else
57 {
58 // Add new entry
59 Services::FeedingScheduleEntry newEntry;
60 newEntry._min = timeMinutesAfterMidnight;
61 newEntry._id = slotIndex;
62 newEntry._dose = dose;
63 newEntry._enabled = enabled;
64 scheduleList.push_back(newEntry);
65 }
66
67 // Save updated schedule back to storage
68 return Set<Services::FeeddingScheduleList>(
69 Services::FieldId::FEEDING_SCHEDULE,
70 scheduleList
71 );

151 else

161bool StorageService::LoadConfigInternal()
162{
163 CORE_INFO("Loading config from EEPROM...");
164
165 std::vector<uint8_t> buffer(MAX_CONFIG_SIZE);
166 if (!_eepromMemory->ReadBytes(CONFIG_START_ADDR, buffer.data(), MAX_CONFIG_SIZE))
167 {
168 return false;
169 }
170
171 buffer.back() = '\0';
172
173 std::string jsonStr(reinterpret_cast<char*>(buffer.data()));
174
175 if (jsonStr.empty() || jsonStr[0] == (char)0xFF)
176 {
177 CORE_INFO("EEPROM appears empty or uninitialized.");
178 return false;
179 }
180
181 if (_configCache.FromJson(jsonStr))
182 {
183 return true;
184 }
185 else
186 {
187 CORE_ERROR("Failed to parse JSON from EEPROM.");
188 return false;
189 }
190}
