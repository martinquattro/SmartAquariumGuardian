23bool EepromMemory::WriteBytes(uint16_t address, const uint8_t* data, size_t length)
24{
25 if (address + length > EEPROM_SIZE_BYTES)
26 {
27 CORE_ERROR("Write out of bounds");
28 return false;
29 }
30
31 uint16_t currentAddress = address;
32 size_t bytesWritten = 0;
33
34 while (bytesWritten < length)
35 {
36 uint16_t pageOffset = currentAddress % BYTES_PER_PAGE;
37 size_t bytesCanWrite = BYTES_PER_PAGE - pageOffset;
38 size_t bytesToWrite = std::min(bytesCanWrite, length - bytesWritten);
39
40 if (!WritePageInternal(currentAddress, &data[bytesWritten], bytesToWrite))
41 {
42 return false;
43 }
44
45 currentAddress += bytesToWrite;
46 bytesWritten += bytesToWrite;
47 }
48
49 return true;
50}

53bool EepromMemory::ReadBytes(uint16_t address, uint8_t* buffer, size_t length)
54{
55 if (address + length > EEPROM_SIZE_BYTES)
56 {
57 CORE_ERROR("Read out of bounds");
58 return false;
59 }
60
61 uint8_t addrPtr[2];
62 addrPtr[0] = (address >> 8) & 0xFF;
63 addrPtr[1] = address & 0xFF;
64
65 if (!_i2c.Write(addrPtr, 2))
66 {
67 CORE_ERROR("Failed to set read address");
68 return false;
69 }
70
71 if (!_i2c.Read(buffer, length))
72 {
73 CORE_ERROR("Failed to read data");
74 return false;
75 }
76 return true;
77}

79//----private------------------------------------------------------------------
80bool EepromMemory::WritePageInternal(uint16_t memAddress, const uint8_t* data, size_t length)
81{
82 std::vector<uint8_t> buffer(2 + length);
83 buffer[0] = (memAddress >> 8) & 0xFF;
84 buffer[1] = memAddress & 0xFF;
85 memcpy(&buffer[2], data, length);
86
87 if (!_i2c.Write(buffer.data(), buffer.size()))
88 {
89 CORE_ERROR("Write failed");
90 return false;
91 }
92
93 TaskDelayMs(10);
94 return true;
95}
