258//----IStorageService-----------------------------------------------------------
259auto GuardianProxy::SaveFeedingScheduleInStorage(const int timeMinutesAfterMidnight, const int slotIndex, const int dose, const bool enabled) -> bool
260{
261 auto scheduleList = Services::StorageService::GetInstance()->Get<Services::FeeddingScheduleList>(
262 Services::FieldId::FEEDING_SCHEDULE
263 );
264
265 // Find existing entry by slotIndex
266 auto it = std::find_if(
267 scheduleList.begin(),
268 scheduleList.end(),
269 [slotIndex](const Services::FeedingScheduleEntry& entry)
270 {
271 return entry._id == slotIndex;
272 }
273 );
274
275 if (it != scheduleList.end())
276 {
277 // Update existing entry
278 it->_min = timeMinutesAfterMidnight;
279 it->_dose = dose;
280 it->_enabled = enabled;
281 }
282 else
283 {
284 // Add new entry
285 Services::FeedingScheduleEntry newEntry;
286 newEntry._min = timeMinutesAfterMidnight;
287 newEntry._id = slotIndex;
288 newEntry._dose = dose;
289 newEntry._enabled = enabled;
290 scheduleList.push_back(newEntry);
291 }
292
293 // Save updated schedule back to storage
294 return Services::StorageService::GetInstance()->Set<Services::FeeddingScheduleList>(
295 Services::FieldId::FEEDING_SCHEDULE,
296 scheduleList
297 );
298}
