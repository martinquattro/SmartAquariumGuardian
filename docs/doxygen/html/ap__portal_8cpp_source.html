35//----private------------------------------------------------------------------
36void APPortal::OnUpdate()
37{
38 switch (_state)
39 {
40 case State::IDLE:
41 {
42 // nothing to do
43 }
44 break;
45
46 case State::INIT:
47 {
48 _state = _Start();
49 }
50 break;
51
52 case State::LISTENING_WIFI_CONFIG:
53 {
54 // waiting for HTTP handler to set WIFI_CREDENTIALS_RECEIVED
55 }
56 break;
57
58 case State::WIFI_CREDENTIALS_RECEIVED:
59 {
60 // Credentials ready for NetworkController to process
61 // Waiting for NetworkController to call ResetState() after processing
62 }
63 break;
64
65 case State::STOP:
66 {
67 _state = _Stop();
68 }
69 break;
70
71 case State::ERROR:
72 {
73 // Error state - could implement retry logic or require manual reset
74 CORE_ERROR("APPortal is in ERROR state");
75 }
76 break;
77 }
78}

145//----private------------------------------------------------------------------
146void APPortal::EnableAPMode()
147{
148 if (_netif == nullptr)
149 {
150 _netif = esp_netif_create_default_wifi_ap();
151 }
152
153 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
154 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
155
156 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_APSTA));
157
158 wifi_config_t wifi_config = {};
159 std::strncpy(reinterpret_cast<char*>(wifi_config.ap.ssid), AP_SSID, sizeof(wifi_config.ap.ssid) - 1);
160 std::strncpy(reinterpret_cast<char*>(wifi_config.ap.password), AP_PASSWORD, sizeof(wifi_config.ap.password) - 1);
161
162 wifi_config.ap.ssid_len = std::strlen(AP_SSID);
163 wifi_config.ap.channel = 1;
164 wifi_config.ap.max_connection = 4;
165 wifi_config.ap.authmode = WIFI_AUTH_WPA_WPA2_PSK;
166
167 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &wifi_config));
168
169 // Configure IP address for AP
170 esp_netif_ip_info_t ipInfo;
171 IP4_ADDR(&ipInfo.ip, 192, 168, 4, 1);
172 IP4_ADDR(&ipInfo.gw, 192, 168, 4, 1);
173 IP4_ADDR(&ipInfo.netmask, 255, 255, 255, 0);
174
175 esp_netif_dhcps_stop(_netif);
176 ESP_ERROR_CHECK(esp_netif_set_ip_info(_netif, &ipInfo));
177 esp_netif_dhcps_start(_netif);
178
179 ESP_ERROR_CHECK(esp_wifi_start());
180
181 CORE_INFO("AP mode enabled: SSID=%s, IP=%s", AP_SSID, AP_IP);
182}

194void APPortal::SetupHttpServer()
195{
196 httpd_config_t config = HTTPD_DEFAULT_CONFIG();
197 config.server_port = HTTP_PORT;
198 config.max_uri_handlers = 6;
199 config.stack_size = 10240;
200
201 if (httpd_start(&_httpServer, &config) != ESP_OK)
202 {
203 CORE_ERROR("APPortal: Failed to start HTTP server");
204 _state = State::ERROR;
205 return;
206 }
207
208 // Register URI handlers
209 httpd_uri_t uri_get_root = {
210 .uri = "/",
211 .method = HTTP_GET,
212 .handler = HandleGetRoot,
213 .user_ctx = this
214 };
215 httpd_register_uri_handler(_httpServer, &uri_get_root);
216
217 httpd_uri_t uri_get_networks = {
218 .uri = "/api/wifi-networks",
219 .method = HTTP_GET,
220 .handler = HandleGetWifiNetworks,
221 .user_ctx = this
222 };
223 httpd_register_uri_handler(_httpServer, &uri_get_networks);
224
225 httpd_uri_t uri_post_credentials = {
226 .uri = "/api/credentials",
227 .method = HTTP_POST,
228 .handler = HandlePostCredentials,
229 .user_ctx = this
230 };
231 httpd_register_uri_handler(_httpServer, &uri_post_credentials);
232
233 httpd_uri_t uri_get_status = {
234 .uri = "/api/status",
235 .method = HTTP_GET,
236 .handler = HandleGetStatus,
237 .user_ctx = this
238 };
239 httpd_register_uri_handler(_httpServer, &uri_get_status);
240
241 CORE_INFO("HTTP server started on port %d", HTTP_PORT);
242}

263//----private------------------------------------------------------------------
264esp_err_t APPortal::HandlePostCredentials(httpd_req_t* req)
265{
266 APPortal* instance = Connectivity::APPortal::GetInstance();
267
268 // Read request body
269 char buf[512] = {0};
270 int total_len = req->content_len;
271 int cur_len = 0;
272
273 while (cur_len < total_len)
274 {
275 int recv_len = httpd_req_recv(req, buf + cur_len, total_len - cur_len);
276 if (recv_len <= 0)
277 {
278 httpd_resp_send_500(req);
279 return ESP_FAIL;
280 }
281 cur_len += recv_len;
282 }
283 buf[cur_len] = '\0';
284
285 // Parse JSON using nlohmann::json
286 Json response;
287 {
288 Json j = Json::parse(std::string(buf), nullptr, false);
289
290 if (j.is_discarded())
291 {
292 response["success"] = false;
293 response["message"] = "Invalid JSON";
294 std::string json_str = response.dump();
295 httpd_resp_set_type(req, "application/json");
296 httpd_resp_send(req, json_str.c_str(), json_str.length());
297 return ESP_OK;
298 }
299
300 std::string ssid, password;
301
302 // Extract SSID
303 if (j.contains("ssid") && j["ssid"].is_string())
304 {
305 ssid = j["ssid"].get<std::string>();
306 }
307
308 // Extract password
309 if (j.contains("password") && j["password"].is_string())
310 {
311 password = j["password"].get<std::string>();
312 }
313
314 if (ssid.empty() || password.empty())
315 {
316 response["success"] = false;
317 response["message"] = "Invalid credentials";
318 std::string json_str = response.dump();
319 httpd_resp_set_type(req, "application/json");
320 httpd_resp_send(req, json_str.c_str(), json_str.length());
321 return ESP_OK;
322 }
323
324 // Store credentials
325 instance->_configuredSsid = ssid;
326 instance->_configuredPassword = password;
327 instance->_state = State::WIFI_CREDENTIALS_RECEIVED;
328
329 CORE_INFO("APPortal: WiFi credentials received - SSID: %s", ssid.c_str());
330
331 response["success"] = true;
332 response["message"] = "Credentials saved. Connecting...";
333 }
334
335 std::string json_str = response.dump();
336 httpd_resp_set_type(req, "application/json");
337 httpd_resp_send(req, json_str.c_str(), json_str.length());
338
339 return ESP_OK;
340}

342//----private------------------------------------------------------------------
343esp_err_t APPortal::HandleGetWifiNetworks(httpd_req_t* req)
344{
345 // Perform WiFi scan
346 esp_wifi_scan_start(nullptr, true); // Blocking scan
347
348 // Get scan results
349 uint16_t apCount = 0;
350 esp_wifi_scan_get_ap_num(&apCount);
351
352 Json response;
353 response["networks"] = Json::array();
354
355 if (apCount > 0)
356 {
357 // Allocate memory for results
358 wifi_ap_record_t* apRecords = new wifi_ap_record_t[apCount];
359 esp_wifi_scan_get_ap_records(&apCount, apRecords);
360
361 // Build JSON response
362 for (uint16_t i = 0; i < apCount; i++)
363 {
364 Json network;
365
366 // Convert SSID to string (null-terminated)
367 std::string ssid(reinterpret_cast<const char*>(apRecords[i].ssid));
368
369 network["ssid"] = ssid;
370 network["rssi"] = apRecords[i].rssi;
371 network["channel"] = apRecords[i].primary;
372
373 response["networks"].push_back(network);
374 }
375
376 delete[] apRecords;
377 }
378
379 std::string json_str = response.dump();
380 httpd_resp_set_type(req, "application/json");
381 httpd_resp_send(req, json_str.c_str(), json_str.length());
382
383 return ESP_OK;
384}
