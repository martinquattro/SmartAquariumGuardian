16/**
17 * @brief Base abstract class for all modules in the system.
18 *
19 * Provides the common lifecycle management:
20 * - Init(): Initialize the module (calls OnInit internally)
21 * - Update(): Periodic update (calls OnUpdate internally, optional)
22 * - GetModuleName(): Get human-readable module name for logging
23 *
24 * All modules (Managers, Drivers, Services) inherit from this.
25 *
26 * @note The parent handles "Initializing <name>" logging automatically.
27 * Derived classes just need to implement OnInit(), OnUpdate() (optional).
28 *
29 */
30class Module
31{
32 public:
33
34 /**
35 * @brief Virtual destructor.
36 */
37 virtual ~Module() = default;
38
39 /**
40 * @brief Initialize the module.
41 * Prints "Initializing <module_name>..." automatically.
42 * Calls OnInit() internally, then optionally delays.
43 * @param delayAfterMs Delay in milliseconds to apply after OnInit() (default 0).
44 * @return bool True if initialization successful, false otherwise.
45 */
46 bool Init(int delayAfterMs = 0)
47 {
48 CORE_INFO("%s Initializing...", GetModuleName());
49
50 if (!OnInit())
51 {
52 CORE_ERROR("%s Failed to initialize", GetModuleName());
53 return false;
54 }
55
56 CORE_INFO("%s initialized successfully", GetModuleName());
57
58 if (delayAfterMs > 0)
59 {
60 TaskDelayMs(delayAfterMs);
61 }
62
63 return true;
64 }

73 OnUpdate();
