18/**
19 * @brief Global registry for module instances.
20 *
21 * This registry allows any module to be retrieved by type anywhere in the system.
22 * Each derived module class (Manager, Driver, Service) registers itself upon
23 * initialization, making it accessible via GetInstance<T>().
24 *
25 * Example:
26 * @code
27 * // In MyManager initialization
28 * SingletonRegistry::Register<MyManager>(this);
29 *
30 * // Later, anywhere in the code
31 * auto* manager = SingletonRegistry::Get<MyManager>();
32 * @endcode
33 */
34class SingletonRegistry
35{
36 public:
37
38 /**
39 * @brief Register a module instance.
40 * @tparam T The derived module class
41 * @param instance Pointer to the module instance
42 */
43 template <typename T>
44 static void Register(T* instance)
45 {
46 GetRegistry()[typeid(T).name()] = static_cast<Module*>(instance);
47 }
48
49 /**
50 * @brief Retrieve a registered module instance by type.
51 * @tparam T The derived module class
52 * @return T* Pointer to the instance, or nullptr if not registered
53 */
54 template <typename T>
55 static T* Get()
56 {
57 auto it = GetRegistry().find(typeid(T).name());
58 if (it != GetRegistry().end())
59 {
60 return static_cast<T*>(it->second);
61 }
62 return nullptr;
63 }
64
65 /**
66 * @brief Check if a module type is registered.
67 * @tparam T The derived module class
68 * @return bool True if registered, false otherwise
69 */
70 template <typename T>
71 static bool IsRegistered()
72 {
73 return GetRegistry().find(typeid(T).name()) != GetRegistry().end();
74 }
75
76 /**
77 * @brief Unregister a module instance (cleanup).
78 * @tparam T The derived module class
79 */
80 template <typename T>
81 static void Unregister()
82 {
83 GetRegistry().erase(typeid(T).name());
84 }
85
86 private:
87
88 /**
89 * @brief Global registry storing all module instances.
90 */
91 static std::map<const char*, Module*>& GetRegistry()
92 {
93 static std::map<const char*, Module*> registry;
94 return registry;
95 }
96
97 // Prevent instantiation
98 SingletonRegistry() = delete;
99 ~SingletonRegistry() = delete;
100};
